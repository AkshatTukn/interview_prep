class Solution {
public:
    int missingNumber(vector<int>& nums) 
    {
       int n=nums.size();
        int sum=0;
        for(int x:nums)
        {
            sum+=x;
        }
      int cur= (n*(n+1))/2;  
       if(cur==sum)
       {
           return 0;
       }
        
       return cur-sum; 
        
    }
};
